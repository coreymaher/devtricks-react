Why use React?

Tell the interpreter what to do


function onClick(e) {
    document.querySelectorAll('#nav .selected').forEach((el) {
        el.classList.remove('selected');
    });

    e.target.classList.add('selected');
}

Declarative

Describe the result you want


function render() {
    return (
        <nav>

Rendering Lists


const users = [
    { name: 'Malcolm Reynolds', occupation: 'Captain / Browncoat' },
    { name: 'Svlad Cjelli', occupation: 'Holistic Detective' },
    { name: 'Daniel Jackson', occupation: 'Archaeologist' },
];
const userElements = users.map((user) => {
    return <div>{user.name} - {user.occupation}</div>;
});
const element = <div>
    <h1>Users</h1>
    <hr />
    <div>{userElements}</div>
</div>;
                    
Users
Malcolm Reynolds - Captain / Browncoat
Svlad Cjelli - Holistic Detective
Daniel Jackson - Archaeologist

Rendering Lists - Keys


const userElements = users.map((user) => {
    return <div>{user.name} - {user.occupation}</div>;
});
                    

Warning: Each child in an array or iterator
should have a unique "key" prop.
                    

const userElements = users.map((user) => {
    return <div key={user.name}>{user.name} - {user.occupation}</div>;
});
